**Part A: Add session_attach() function to bin/wez-session**

Add a `session_attach()` function to bin/wez-session, placed after `session_delete()` and before the daemon functions. Implement smart attach behavior entirely in bash (the Lua restore API handles the complex logic when called from WezTerm's Lua context, but CLI needs its own path).

The CLI attach strategy (per CONTEXT.md "CLI bridge mechanism" being Claude's discretion):

```bash
session_attach() {
    local name="$1"

    if [[ -z "$name" ]]; then
        echo "Error: Session name required. Usage: wez-session attach <name>" >&2
        exit 1
    fi

    ensure_mux_running

    # Check if workspace is already running
    local active_workspaces=$(get_active_workspaces)
    if echo "$active_workspaces" | grep -q "^${name}$"; then
        # Already running -- spawn window in that workspace to switch to it
        wezterm cli spawn --new-window --workspace "$name" >/dev/null 2>&1
        echo "Attached to running session: ${name}"
        return 0
    fi

    # Not running -- check if saved JSON exists
    local json_file="$SESSIONS_DIR/${name}.json"
    if [[ ! -f "$json_file" ]]; then
        echo "Error: Session '${name}' not found (no active workspace, no saved state)" >&2
        exit 1
    fi

    # Restore from JSON using CLI commands
    echo "Restoring session '${name}' from saved state..."

    # Parse JSON and restore tabs/panes
    # Strategy: spawn new workspace, then create tabs and splits from JSON
    local restore_result
    restore_result=$(python3 -c "
import json, subprocess, sys, os

json_file = '$json_file'
name = '${name}'
home = os.path.expanduser('~')

with open(json_file) as f:
    layout = json.load(f)

tabs = layout.get('tabs', [])
if not tabs:
    print('Error: No tabs in saved session', file=sys.stderr)
    sys.exit(1)

# Restorable processes (shells excluded -- they spawn by default)
RESTORABLE = {
    'nvim': 'nvim',
    'claude': 'claude',
    'npm': 'npm run dev',
    'node': 'node',
}

def run_cmd(args, check=True):
    \"\"\"Run a command and return stdout, or None on failure.\"\"\"
    try:
        result = subprocess.run(args, capture_output=True, text=True, timeout=10)
        if check and result.returncode != 0:
            return None
        return result.stdout.strip()
    except Exception as e:
        print(f'Warning: Command failed: {args}: {e}', file=sys.stderr)
        return None

def get_cwd(pane_data):
    \"\"\"Get CWD from pane data, falling back to home dir.\"\"\"
    cwd = pane_data.get('cwd') or home
    if not os.path.isdir(cwd):
        print(f'Warning: CWD not found, using home: {cwd}', file=sys.stderr)
        cwd = home
    return cwd

def infer_direction(prev_pane, curr_pane):
    \"\"\"Infer split direction from geometry.\"\"\"
    prev_left = prev_pane.get('left', 0)
    prev_width = prev_pane.get('width', 80)
    curr_left = curr_pane.get('left', 0)
    if curr_left > prev_left + (prev_width / 2):
        return '--right'
    return '--bottom'

def send_command(pane_id, process):
    \"\"\"Send command to pane if process is restorable.\"\"\"
    if not process:
        return
    # Skip shells
    if process in ('zsh', 'bash', 'fish', 'sh', 'dash'):
        return
    cmd = RESTORABLE.get(process)
    if cmd:
        run_cmd(['wezterm', 'cli', 'send-text', '--pane-id', str(pane_id), '--no-paste', cmd + chr(10)], check=False)
    else:
        print(f'Skipped process restoration: {process}', file=sys.stderr)

# Step 1: Create new workspace with first pane's CWD
first_cwd = get_cwd(tabs[0]['panes'][0]) if tabs[0].get('panes') else home
initial_pane_id = run_cmd(['wezterm', 'cli', 'spawn', '--new-window', '--workspace', name, '--cwd', first_cwd])

if not initial_pane_id:
    print('Error: Failed to create workspace', file=sys.stderr)
    sys.exit(1)

# Configure first pane (send command if needed)
if tabs[0].get('panes'):
    send_command(initial_pane_id, tabs[0]['panes'][0].get('process'))

# Set tab title for first tab
title = tabs[0].get('title', '')
if title:
    run_cmd(['wezterm', 'cli', 'set-tab-title', '--pane-id', initial_pane_id, title], check=False)

# Step 2: Create remaining panes in first tab
if tabs[0].get('panes') and len(tabs[0]['panes']) > 1:
    panes = tabs[0]['panes']
    for i in range(1, len(panes)):
        direction = infer_direction(panes[i-1], panes[i])
        cwd = get_cwd(panes[i])
        new_pane_id = run_cmd(['wezterm', 'cli', 'split-pane', direction, '--pane-id', initial_pane_id, '--cwd', cwd])
        if new_pane_id:
            send_command(new_pane_id, panes[i].get('process'))
        else:
            print(f'Warning: Failed to create pane {i} in tab 1', file=sys.stderr)
        if i > 4:
            print(f'Warning: Layout accuracy may degrade beyond 4 panes', file=sys.stderr)

# Step 3: Create additional tabs
for tab_idx in range(1, len(tabs)):
    tab_data = tabs[tab_idx]
    panes = tab_data.get('panes', [])
    if not panes:
        continue

    first_cwd = get_cwd(panes[0])
    tab_pane_id = run_cmd(['wezterm', 'cli', 'spawn', '--pane-id', initial_pane_id, '--cwd', first_cwd])

    if not tab_pane_id:
        print(f'Warning: Failed to create tab {tab_idx + 1}', file=sys.stderr)
        continue

    # Send command to first pane
    send_command(tab_pane_id, panes[0].get('process'))

    # Set tab title
    title = tab_data.get('title', '')
    if title:
        run_cmd(['wezterm', 'cli', 'set-tab-title', '--pane-id', tab_pane_id, title], check=False)

    # Create splits for remaining panes
    for i in range(1, len(panes)):
        direction = infer_direction(panes[i-1], panes[i])
        cwd = get_cwd(panes[i])
        new_pane_id = run_cmd(['wezterm', 'cli', 'split-pane', direction, '--pane-id', tab_pane_id, '--cwd', cwd])
        if new_pane_id:
            send_command(new_pane_id, panes[i].get('process'))
        else:
            print(f'Warning: Failed to create pane {i} in tab {tab_idx + 1}', file=sys.stderr)

print(f'Restored {len(tabs)} tab(s) from saved state')
" 2>&1)

    local exit_code=$?
    if [[ $exit_code -ne 0 ]]; then
        echo "Error: Restoration failed" >&2
        echo "$restore_result" >&2
        exit 1
    fi

    echo "$restore_result"
    echo "Restored session: ${name}"
}
```
